\documentclass{article}

\usepackage{siunitx}
\usepackage[record,stylemods=longbooktabs]{glossaries-extra}

\newglossary*{units}{Units of Measurement (by SI unit)}
\newglossary*{units2}{Units of Measurement (by category)}

\GlsXtrLoadResources[
  % data in baseunits.bib and derivedunits.bib:
  src={baseunits,derivedunits},
  category={same as base},% set category field to base name
  sort-field={name},% sort by name field
  selection=all,% select all entries
  type=main,% put primary entries in main glossary
  % make @symbol act like @dualsymbol:
  entry-type-aliases={symbol=dualsymbol},
  set-widest,% needed for alttree style
  dual-sort={letternumber-upperlower},
  %dual-letter-number-punc-rule={punc-first-space-zero},
  dual-type={units}
]

\GlsXtrLoadResources[
  src={baseunits,derivedunits},
  selection={all},
  category={same as base},
  sort-field={category},
  identical-sort-action={name},
  sort={letter-case},
  type=units2,
  action=copy
]

\setlength{\glsdescwidth}{.4\hsize}

% define custom glossary style
\newglossarystyle{units}% style name
{% base it on long3col-booktabs
  \setglossarystyle{long3col-booktabs}%
  \renewcommand*{\glossaryheader}{%
    \toprule 
    \bfseries Name &
    \bfseries Measurement &
    \bfseries Symbol
    \tabularnewline\midrule\endhead
    \bottomrule\endfoot}%
% main entries:
  \renewcommand{\glossentry}[2]{%
    \glsentryitem{##1}\glstarget{##1}{\glossentryname{##1}} &
    \glossentrydesc{##1}\glspostdescription &
    \glossentrysymbol{##1}\tabularnewline
  }%
% no sub-entries, but define for completeness:
  \renewcommand{\subglossentry}[3]{%
     \glssubentryitem{##2}%
     \glstarget{##2}{\glossentryname{##2}}&
     \glossentrydesc{##2}\glspostdescription &
     \glossentrysymbol{##2}\tabularnewline
  }%
}

\printunsrtglossary[title={SI Units of Measurement},
 style={units}]

\printunsrtglossary[type=units,style=alttree]

\printunsrtglossary*[type=units2,style=indexgroup]
 {\renewcommand{\glsxtrgroupfield}{category}}
